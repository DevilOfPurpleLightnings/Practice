MySql

Для закріплення навичок роботи з реляційними базами даних я виконав серію SQL-запитів. У процесі роботи було використано вибірки даних, вкладені підзапити та механізми сортування результатів.

Робота з базовими вибірками

На першому етапі було виконано запити для отримання даних із використанням підзапитів. Зокрема, визначались міста без вулиць, категорії товарів із продажами та товари, ціна яких перевищує середню у своїй категорії.

SQL 
-- Міста без вулиць
SELECT id, name
FROM city c
WHERE NOT EXISTS (
    SELECT 1 FROM street s WHERE s.city_id = c.id
)
ORDER BY name ASC;

-- Категорії, де всі товари мають покупки
SELECT pc.id, pc.name
FROM product_category pc
WHERE EXISTS (
    SELECT 1 FROM product_title pt WHERE pt.product_category_id = pc.id
)
AND NOT EXISTS (
    SELECT 1
    FROM product_title pt
    JOIN product p ON p.product_title_id = pt.id
    WHERE pt.product_category_id = pc.id
    AND NOT EXISTS (
        SELECT 1 FROM order_details od WHERE od.product_id = p.id
    )
)
ORDER BY pc.id ASC;

-- Товари дорожчі за середню ціну в категорії
SELECT 
    p.id, 
    CAST(p.comment AS CHAR) AS title, 
    p.price
FROM product p
JOIN product_title pt ON p.product_title_id = pt.id
WHERE p.price > (
    SELECT AVG(p2.price)
    FROM product p2
    JOIN product_title pt2 ON p2.product_title_id = pt2.id
    WHERE pt2.product_category_id = pt.product_category_id
)
ORDER BY p.id ASC;

Аналіз покупок клієнтів

Далі було реалізовано запит для визначення клієнтів із середнім чеком понад 70. Для цього застосовано агрегатні функції та групування даних.

SQL 
SELECT 
    p.name, 
    p.surname, 
    SUM(od.product_amount * od.price_with_discount) / COUNT(DISTINCT od.customer_order_id) AS avg_purchase,
    SUM(od.product_amount * od.price_with_discount) AS sum_purchase
FROM customer c
LEFT JOIN person p ON c.person_id = p.id
JOIN customer_order co ON co.customer_id = c.person_id OR (co.customer_id IS NULL AND c.person_id IS NULL)
JOIN order_details od ON od.customer_order_id = co.id
GROUP BY p.name, p.surname
HAVING avg_purchase > 70
ORDER BY sum_purchase ASC, p.surname ASC;

Аналіз впливу знижок

Було досліджено вплив знижок на кількість продажів. Порівнювалась кількість товарів, проданих зі знижкою понад 5% та без неї.

SQL 
SELECT 
    p.id, 
    pt.title,
    SUM(CASE WHEN (od.price - od.price_with_discount) > od.price * 0.05 
        THEN od.product_amount ELSE 0 END) AS count_with_discount_5,
    SUM(CASE WHEN (od.price - od.price_with_discount) <= od.price * 0.05 
        THEN od.product_amount ELSE 0 END) AS count_without_discount_5,
    ROUND(
        (
            SUM(CASE WHEN (od.price - od.price_with_discount) > od.price * 0.05 THEN od.product_amount ELSE 0 END) -
            SUM(CASE WHEN (od.price - od.price_with_discount) <= od.price * 0.05 THEN od.product_amount ELSE 0 END)
        ) * 100.0 /
        NULLIF(SUM(CASE WHEN (od.price - od.price_with_discount) <= od.price * 0.05 THEN od.product_amount ELSE 0 END), 0)
    , 2) AS difference
FROM product p
JOIN product_title pt ON p.product_title_id = pt.id
LEFT JOIN order_details od ON od.product_id = p.id
GROUP BY p.id, pt.title
ORDER BY p.id ASC;

Визначення провідних виробників

Завершальним етапом стало визначення виробника, продукція якого продається найкраще для кожного типу товару.

SQL 
SELECT 
    pt.id AS product_id,
    pt.title,
    m.id AS manufacturer_id,
    m.name AS manufacturer
FROM product_title pt
LEFT JOIN manufacturer m ON m.id = (
    SELECT p.manufacturer_id
    FROM product p
    JOIN order_details od ON p.id = od.product_id
    WHERE p.product_title_id = pt.id
    GROUP BY p.manufacturer_id
    ORDER BY SUM(od.product_amount) DESC
    LIMIT 1
)
ORDER BY pt.id ASC;
